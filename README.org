def create_hex_grid(data : gpd.GeoDataFrame, cell_area : float, trim_mode :str = "clip" ) -> gpd.GeoSeries:
    # Calculate hexagon side from area: A = (3 * sqrt(3) * s^2) / 2
    # circum radius s = R, this is the one we want
    radius = np.sqrt(cell_area * 2 / (3 * np.sqrt(3)))
    utm_crs = data.estimate_utm_crs()
    grid = ghg.make_grid_from_gdf(data.to_crs(utm_crs), radius, trim_mode=trim_mode)
    return grid.to_crs(data.crs)

def estimate_random_sample_points(data : gpd.GeoDataFrame, num_samples : int, buffer_size : float = - 18) -> gpd.GeoDataFrame:
    original_crs = data.crs
    utm_crs = data.estimate_utm_crs()

    geometry = data.to_crs(utm_crs)[["geometry"]]
    new_geometry = geometry.buffer(buffer_size)

    reduced_data = gpd.GeoDataFrame(geometry=new_geometry, crs = new_geometry.crs)
    grid = create_hex_grid(reduced_data, cell_area = 0.25*acre_to_meter)
    cells_to_sample = grid.sample(num_samples, random_state=42)
    points = gpd.GeoSeries(cells_to_sample.sample_points(1, rng = np.random.default_rng(seed=42)).union_all(),crs=utm_crs).to_crs(original_crs)
    data[["random_sample_locations"]] = points

    return data

def estimate_grid_sample_points(data : gpd.GeoDataFrame, cell_area : float, buffer_size : float = - 18) -> gpd.GeoDataFrame:
    original_crs = data.crs
    utm_crs = data.estimate_utm_crs()

    geometry = data.to_crs(utm_crs)[["geometry"]]
    new_geometry = geometry.buffer(buffer_size)

    reduced_data = gpd.GeoDataFrame(geometry=new_geometry, crs = new_geometry.crs)
    grid = create_hex_grid(reduced_data, cell_area = cell_area*acre_to_meter, trim_mode="intersect")
    #grid = shapely.geometry.MultiPolygon(grid.geometry.values)
    #grid_geom = gpd.GeoSeries(grid, crs=utm_crs)
    grid =  gpd.clip(grid, new_geometry)

    points = gpd.GeoSeries(grid.sample_points(1, rng = np.random.default_rng(seed=42)).union_all(),crs=utm_crs).to_crs(original_crs)
    point_rep = gpd.GeoSeries(grid.representative_point().union_all(),crs=utm_crs).to_crs(original_crs)

    # we would like to save the grid too, however some of the grid elements are multipolygon
    # this occurs because of the clip above.
    grid = grid.to_crs(original_crs)

    part_outside_regular_grid = grid.loc[[isinstance(ii,shapely.MultiPolygon) for ii in grid.geometry ],:].explode().geometry.to_list()
    part_of_regular_grid = grid.loc[[isinstance(ii,shapely.Polygon) for ii in grid.geometry ],:].geometry.to_list()
    grid_geom_combined = shapely.geometry.MultiPolygon(part_outside_regular_grid + part_of_regular_grid)

    data[[f"grid_random_sample_locations_{cell_area}"]] = points
    data[[f"grid_representative_sample_locations_{cell_area}"]] = point_rep
    data[[f"grid_locations_{cell_area}"]] = grid_geom_combined

    return data


def dbeaver_json_to_geojson(fname : str) -> str:
    """
    Take the json output from the query involving:
    select jsonb_build_object(
        'type','FeatureCollection',
        'features', jsonb_agg(st_asgeojson(tt.*)::json) ) info ) from (...) tt
    where tt.features is a column that contains the geometries, while
    the other columns in tt will be properties.
    Why this is needed: saving the output as a json, using DBeaver,
    results in some additional information around the data we need.
    This functions extracts the needed data,
    discards the additional information, and creates a valid geojson.
    """
    outname = os.path.splitext(fname)[0] + ".geojson"
    with open(fname,'r') as fh:
        data = json.load(fh)
        for _,(info_dict,*rest) in data.items():
            # we only take the first key:value pair, the value is a list.
            # we need the first element of this list, which is a dict
            # the dict has only one key, the value to this key is a str encoded json
            for _,info in info_dict.items():
                break
            break
    df = gpd.read_file(StringIO(info))
    df.to_file(outname, driver="GeoJSON")
    return outname

* Analysis

To identify potential target fields for soil sampling we use a clustering analysis. We are looking for groups of co-located fields that are far enough from existing data locations. All together, we leverage a clustering approach where we cluster as many fields together as we can. This implies that an optimization strategy would good to use.

** Clustering to identify target locations

Cluster the combined data sample locations.

#+begin_src python :session soilgpt-sample :results value :pandoc t
# data = soilgpt_sample_locations.set_index("sample_id")
data = combined_data_with_meta.to_pandas().set_index("field_id")
def objective(data, trial):
    N = len(data)
    min_cluster_size = trial.suggest_int('min_cluster_size', 3,50)
    min_samples = trial.suggest_int('min_samples', 3,50)
    cluster_model = HDBSCAN(min_cluster_size=min_cluster_size, min_samples=min_samples).fit(data[["s1","s2","s3"]])
    noise_size = float(sum(cluster_model.labels_ == -1))
    return noise_size/N
obj_function = partial(objective, data)

For the random sample we partition the field into 0.25 acre grids, then select a random set of cells equal to the sample size requirements, and finally a random point inside the respective cells.
